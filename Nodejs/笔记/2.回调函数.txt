阻塞式：先输出数据，后打印信息
var fs = require("fs");//引入包（模块）fs
var data = fs.readFileSync('input.txt');		//readFileSync 函数
console.log(data.toString());
console.log("done!");

非阻塞式：先打印信息，后输出数据
var fs = require("fs");
fs.readFile('input.txt', function (err, data) {		//readFile 函数
    if (err) return console.error(err);
    console.log(data.toString());
});
console.log("done!");	//先输出
